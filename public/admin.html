const express = require("express"); const mongoose = require("mongoose"); const bodyParser = require("body-parser"); const path = require("path"); const app = express(); // Middleware app.use(bodyParser.json()); app.use(express.static("public")); /* ========================= MongoDB Atlas Connection ========================= */ // Note: Password sahi check kar lena (grocery:grocery) mongoose.connect("mongodb+srv://groceryuser:grocery123@cluster0.6xciiec.mongodb.net/groceryDB?retryWrites=true&w=majority") .then(() => console.log("MongoDB Connected ✅")) .catch(err => console.log("Mongo Error: ", err)); /* ========================= Schemas (Database Design) ======================== */ // 1. User Schema (Registration/Login ke liye) const userSchema = new mongoose.Schema({ name: String, email: String, password: String, date: { type: Date, default: Date.now } }); // 2. Product Schema (Inventory Manage karne ke liye) const productSchema = new mongoose.Schema({ name: String, price: Number, category: String, image: String, description: String }); // 3. Order Schema (Professional details ke saath) const orderSchema = new mongoose.Schema({ userName: String, userEmail: String, address: String, phone: String, items: Array, total: Number, date: { type: Date, default: Date.now } }); const User = mongoose.model("User", userSchema); const Product = mongoose.model("Product", productSchema); const Order = mongoose.model("Order", orderSchema); /* ========================= API Routes (Back-end logic) ========================= */ // --- 1. Products Fetch (Home page par dikhane ke liye) --- app.get("/products", async (req, res) => { try { const products = await Product.find(); res.json(products); } catch (err) { res.status(500).json({ error: "Products load nahi ho sake" }); } }); // --- 2. Add Product (Admin Panel se save karne ke liye) --- app.post("/addProduct", async (req, res) => { try { const newProduct = new Product(req.body); await newProduct.save(); res.json({ message: "Product Database mein save ho gaya! ✅" }); } catch (err) { res.status(400).json({ error: "Product save nahi hua" }); } }); // --- 3. Delete Product (Admin Panel se hatane ke liye) --- app.delete("/deleteProduct/:id", async (req, res) => { try { await Product.findByIdAndDelete(req.params.id); res.json({ message: "Product successfully deleted! 🗑️" }); } catch (err) { res.status(500).json({ error: "Delete nahi ho paya" }); } }); // --- 4. Save Order (Checkout ke waqt) --- app.post("/order", async (req, res) => { try { const { items, total, userName, address, phone, userEmail } = req.body; const newOrder = new Order({ userName, userEmail, address, phone, items, total }); await newOrder.save(); res.json({ message: "Order Placed Successfully! 🚚" }); } catch (err) { res.status(500).json({ error: "Order process nahi ho saka" }); } }); // --- 5. Admin Data (Dashboard ke analytics ke liye) --- app.get("/adminData", async (req, res) => { try { const users = await User.find(); const orders = await Order.find(); const products = await Product.find(); res.json({ users, orders, products }); } catch (err) { res.status(500).json({ error: "Admin data fetch nahi ho saka" }); } }); /* ========================= Server Start ========================= */ const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`🚀 Server running on http://localhost:${PORT}`); });
